--INNER JOIN: Datos en ambas tablas, tomar registros que esten en comun en ambas tablas.
SELECT NombreCliente, 
       Contrato,
	   Factura,
	   FechaFact,
	   Monto
FROM Clientes 
INNER JOIN Pedidos 
ON Clientes.IdCliente = Pedidos.IdCliente
WHERE Estatus = 'A'
--LEFT JOIN: combinar registros, left la primera tabla y right la segunda
SELECT NombreCliente, 
       Contrato,
	   Factura,
	   FechaFact,
	   Monto
FROM Clientes 
LEFT JOIN Pedidos 
ON Clientes.IdCliente = Pedidos.IdCliente
WHERE Estatus = 'A'
--RIGTH JOIN
SELECT NombreCliente, 
       Contrato,
	   Factura,
	   FechaFact,
	   Monto
FROM Clientes 
RIGHT JOIN Pedidos 
ON Clientes.IdCliente = Pedidos.IdCliente
WHERE Estatus = 'A'
--FULL JOIN
SELECT NombreCliente, 
       Contrato,
	   Factura,
	   FechaFact,
	   Monto
FROM Clientes 
FULL JOIN Pedidos: Tomar ambas tablas
ON Clientes.IdCliente = Pedidos.IdCliente
WHERE Estatus = 'A'
